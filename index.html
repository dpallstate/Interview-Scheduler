const formatTime = (time) => { // Handles both {hour, minute} objects and numbers representing minutes since midnight let hour, minute; if (typeof time === 'object' && time !== null) { hour = time.hour; minute = time.minute; } else if (typeof time === 'number') { hour = Math.floor(time / 60); minute = time % 60; } else { return time; // fallback } const date = new Date(); date.setHours(hour); date.setMinutes(minute); return date.toLocaleString('en-US', { hour: 'numeric', minute: '2-digit', hour12: true }); };
